def initialize(manager, opts = {})
  @options = opts
  @manager = manager
  @self_read, @self_write = IO.pipe
  trap_signals(NUM_TO_SIGNAL, @self_write)
  @worker_manager = WorkerManager.new(create_master_socket)
  at_exit { delete_socket_file }
  @running = true
end
